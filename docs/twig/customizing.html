Customizing Filters and Functions



Filter:

<?php

use Core\Template;

Template::twigFilter('your_filter', function ($data) {
    return 'Your code...';
});

?>

{{ 'My string...'|your_filter }}





Function:

<?php

use Core\Template;

Template::twigFunction('your_function', function ($data) {
    return 'Your code...';
});

?>

{{ your_function('My string...') }}



File: libraries/TwigFilter.php

<?php
/**
 * TwigFilter
 *
 * Demo class for creating custom filters.
 */
namespace App;

use Core\Template;

class TwigFilter
{
    /**
     * __construct
     *
     * @access public
     */
    public function __construct()
    {
        Template::twigFilter('demo_filter', function ($str) {
            return $this->demoFilter($str);
        });
    }

    /**
     * Demo filter
     *
     * @param string $str
     * @return string
     * @access private
     */
    private function demoFilter($str) : string
    {
        return str_replace('...', '!', $str);
    }
}

?>



File: libraries/TwigFunction.php

<?php
/**
 * TwigFunction
 *
 * Demo class for creating custom functions.
 */
namespace App;

use Core\Template;

class TwigFunction
{
    /**
     * __construct
     *
     * @access public
     */
    public function __construct()
    {
        Template::twigFunction('demo_function', function ($str) {
            return $this->demoFunction($str);
        });
    }

    /**
     * Demo function
     *
     * @param string $str
     * @return string
     * @access private
     */
    private function demoFunction($str) : string
    {
        return $str;
    }
}

?>
